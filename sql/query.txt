SELECT sheet1.old_pin, sheet1.owner, sheet2.pin_new
FROM sheet1
INNER JOIN sheet2
ON sheet1.old_pin = sheet2.old_pin
GROUP BY sheet1_id;

BEGIN
    SELECT COUNT(sheet1.old_pin)
	FROM sheet1
    INNER JOIN sheet2
    ON sheet1.old_pin = sheet2.old_pin
    GROUP BY sheet1_id; 
END

SELECT DISTINCT SUBSTRING(sheet1.old_pin,12,2) AS old_pin
FROM sheet1
INNER JOIN sheet2
ON sheet1.old_pin = sheet2.old_pin
GROUP BY sheet1_id;
